# Environment
IS_DEVELOPMENT=string (must be either "true" or "false")

# User Credentials
USER_EMAIL=string (example: johndoe@gmail.com)
USER_PASSWORD = string (example: qwerty123) though you should likely use a strong hashed version

# JWT Configuration
TOKEN_SECRET=string (example: 5ed8a0e881f679121d9550b3530134ae70e457a9ab2f151269d4201e7e86fea6) 
ACCESS_TOKEN_TLL=number (must be greater than 0) *treated as minutes
REFRESH_TOKEN_TTL=number (must be greater than 0) *treated as days

# Database
DB_TYPE=string (example: postgres)
DB_USER=string (example: john)
DB_PASSWORD=string (example: qwerty123)
DB_HOST=string (example: localhost)
DB_PORT=number (example: 5432)
DB_NAME=string (exaple: devdb)

# Logging
LOG_LEVEL=number (must be between [-1, 5])


# Goose
GOOSE_DRIVER=string (example: postgres)
GOOSE_DBSTRING=string (example: postgres://user:password@host:port/db_name)
GOOSE_MIGRATION_DIR= string (example: ./migrations)
